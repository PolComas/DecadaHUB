// Helpers
const mailboxEmail = $('Config').first().json.mailbox_email.toLowerCase();
const mailboxId    = $('Upsert mailbox').first().json.id;

const results = [];

for (const item of $input.all()) {
  const msg = item.json;
  const headers = {};
  for (const h of (msg.payload?.headers ?? [])) {
    headers[h.name.toLowerCase()] = h.value;
  }

  // --- Participants ---
  const parseAddresses = (raw, role) => {
    if (!raw) return [];
    return raw.split(/,(?=(?:[^"]*"[^"]*")*[^"]*$)/).map(addr => {
      const m = addr.trim().match(/^(?:"?([^"<]+)"?\s*)?<?([^>]+)>?$/);
      return {
        email: (m?.[2] ?? addr).trim().toLowerCase(),
        name:  (m?.[1] ?? '').trim(),
        role
      };
    });
  };

  const participants = [
    ...parseAddresses(headers['from'], 'from'),
    ...parseAddresses(headers['to'],   'to'),
    ...parseAddresses(headers['cc'],   'cc')
  ];

  const fromEntry = participants.find(p => p.role === 'from');
  const fromEmail  = fromEntry?.email ?? '';
  const fromDomain = fromEmail.includes('@') ? fromEmail.split('@')[1] : '';

  // --- Direction ---
  const allRecipients = participants
    .filter(p => p.role === 'to' || p.role === 'cc')
    .map(p => p.email);

  let direction;
  if (fromEmail.includes(mailboxEmail)) {
    direction = 'team_to_client';
  } else if (allRecipients.some(e => e.includes(mailboxEmail))) {
    direction = 'client_to_team';
  } else {
    direction = 'internal';
  }

  // --- Clean subject ---
  const rawSubject = headers['subject'] ?? '';
  const cleanSubject = rawSubject
    .replace(/^(Re|RE|Fwd|FWD|Fw|FW):\s*/g, '')
    .trim();

  // --- Extract body_text (plain text preferred) ---
  const extractBody = (payload) => {
    if (!payload) return '';
    const decode = (data) => {
      if (!data) return '';
      try { return atob(data.replace(/-/g, '+').replace(/_/g, '/')); }
      catch { return ''; }
    };
    if (payload.mimeType === 'text/plain' && payload.body?.data) {
      return decode(payload.body.data);
    }
    for (const part of (payload.parts ?? [])) {
      if (part.mimeType === 'text/plain' && part.body?.data) {
        return decode(part.body.data);
      }
    }
    // Fallback: multipart/alternative → recurse
    for (const part of (payload.parts ?? [])) {
      const found = extractBody(part);
      if (found) return found;
    }
    return '';
  };

  const bodyText = extractBody(msg.payload);

  // --- Sent date ---
  const sentAt = headers['date']
    ? new Date(headers['date']).toISOString()
    : new Date(parseInt(msg.internalDate ?? 0)).toISOString();

  results.push({
    json: {
      // IDs from Gmail
      provider_message_id: msg.id,
      provider_thread_id:  msg.threadId,
      history_id:          msg.historyId,
      // Headers
      internet_message_id: headers['message-id'] ?? '',
      references:          headers['references']  ?? '',
      subject:             cleanSubject,
      raw_subject:         rawSubject,
      snippet:             msg.snippet ?? '',
      body_text:           bodyText,
      sent_at:             sentAt,
      direction,
      participants,
      from_email:          fromEmail,
      from_domain:         fromDomain,
      // Context
      mailbox_id:          mailboxId,
      mailbox_email:       mailboxEmail,
      raw_payload:         msg
    }
  });
}

return results;
